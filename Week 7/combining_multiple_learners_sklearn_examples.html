import matplotlib.pyplot as plt
scores = {'train' : [], 'test' : []}

for y_p_train, y_p_test in zip(adaboost.staged_score(X_train, y_train),
                               adaboost.staged_score(X_test, y_test)):
    scores['train'].append(y_p_train)
    scores['test'].append(y_p_test)

# Plot the results.
n_estimators = range(1, len(scores['train']) + 1)
for key in scores.keys():
    plt.plot(n_estimators, scores[key])
plt.title('Staged R^2')
plt.ylabel('R^2')
plt.xlabel('N Estimators')
plt.legend(scores.keys())
plt.show()

import matplotlib.pyplot as plt
import sklearn.metrics as met
scores = {'train' : [], 'test' : []}

for y_predict_train, y_predict_test in zip(GBM.staged_predict(X_train),
                                           GBM.staged_predict(X_test)):
    scores['train'].append(met.r2_score(y_train, y_predict_train))
    scores['test'].append(met.r2_score(y_test, y_predict_test))

# Plot the results.
n_estimators = range(1, len(scores['train']) + 1)
for key in scores.keys():
    plt.plot(n_estimators, scores[key])
plt.title('Staged R^2')
plt.ylabel('R^2')
plt.xlabel('N Estimators')
plt.legend(scores.keys())
